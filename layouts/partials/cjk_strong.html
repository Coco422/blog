{{- /* Fix literal CJK **...** left behind by Goldmark's CommonMark emphasis rules. */ -}} {{- $content := printf "%s" . -}} {{- $scratch := newScratch -}} {{- $scratch.Set "content" $content -}} {{- range $i, $block := findRE `(?s)]*>.*?` $content -}} {{- $key := printf "" $i -}} {{- $scratch.SetInMap "blocks" $key $block -}} {{- $scratch.Set "content" (replace ($scratch.Get "content") $block $key) -}} {{- end -}} {{- $content = $scratch.Get "content" -}} {{- range $i, $block := findRE `(?s)<(?:code|kbd|samp|script|style)\b[^>]*>.*?` $content -}} {{- $key := printf "" $i -}} {{- $scratch.SetInMap "blocks" $key $block -}} {{- $scratch.Set "content" (replace ($scratch.Get "content") $block $key) -}} {{- end -}} {{- $content = $scratch.Get "content" -}} {{- range $i, $tag := findRE `<[^>]+>` $content -}} {{- $key := printf "" $i -}} {{- $scratch.SetInMap "tags" $key $tag -}} {{- $scratch.Set "content" (replace ($scratch.Get "content") $tag $key) -}} {{- end -}} {{- $content = $scratch.Get "content" -}} {{- $cjk := `\p{Han}\p{Hiragana}\p{Katakana}\p{Hangul}，。：；！？、”“‘’「」『』《》（）【】` -}} {{- $content = replaceRE (printf `\*\*\*([^*<>\n]*[%s][^*<>\n]*)\*\*\*` $cjk) `$1` $content -}} {{- $content = replaceRE (printf `\*\*([^*<>\n]*[%s][^*<>\n]*)\*\*` $cjk) `$1` $content -}} {{- range $key, $tag := ($scratch.Get "tags") -}} {{- $content = replace $content $key $tag -}} {{- end -}} {{- range $key, $block := ($scratch.Get "blocks") -}} {{- $content = replace $content $key $block -}} {{- end -}} {{- $content | safeHTML -}}
